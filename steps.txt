1-built the container for kafka service using docker compose file


2-create the topics :
# Créer topic tp8-input
docker exec -it kafka_project-kafka-1 kafka-topics --create \
  --topic tp8-input \
  --bootstrap-server localhost:9092 \
  --partitions 1 \
  --replication-factor 1

# Créer topic tp8-dlq
docker exec -it kafka_project-kafka-1 kafka-topics --create \
  --topic tp8-dlq \
  --bootstrap-server localhost:9092 \
  --partitions 1 \
  --replication-factor 1

# Vérifier les topics créés
docker exec -it kafka_project-kafka-1 kafka-topics --list \
  --bootstrap-server localhost:9092
